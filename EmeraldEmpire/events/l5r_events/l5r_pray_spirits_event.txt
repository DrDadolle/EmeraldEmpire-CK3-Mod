scripted_effect l5r_pray_spirit_0005_get_courtier_who_dont_like = {
	# Checks for a valid rival or nemesis
	if = {
		limit = {
			any_relation = {
				type = nemesis
				is_available_ai_adult = yes
			}
		}
		random_relation = {
			type = nemesis
			limit = {
				is_available_ai_adult = yes
			}
			save_scope_as = enemy
		}
	}
	else_if = {
		limit = {
			any_relation = {
				type = rival
				is_available_ai_adult = yes
			}
		}
		random_relation = {
			type = rival
			limit = {
				is_available_ai_adult = yes
			}
			save_scope_as = enemy
		}
	}
	else = { # grab a random courtier with a bad opinion of you
		random_courtier = {
			limit = {
				opinion = {
					target = root
					value <= -50
				}
			}
			save_scope_as = enemy
		}		
	}
	if = { #grab a random courtier or guest if no enemy
		limit = {
			NOT = { exists = scope:enemy }
		}
		random_courtier = {
			save_scope_as = enemy
		}
	}
}
