scripted_effect resolve_tattoo_effect = {
	if = {
		limit = { scope:tattoo = flag:avalanche}
		send_interface_toast = {
			title = msg_added_tattoo
			left_icon = root
			add_trait = avalanche_tattoo
		}
	}
	else_if = {
		limit = { scope:tattoo = flag:cherry_blossom}
		send_interface_toast = {
			title = msg_added_tattoo
			left_icon = root
			add_trait = cherry_blossom_tattoo
		}
	}
	else_if = {
		limit = { scope:tattoo = flag:volcano}
		send_interface_toast = {
			title = msg_added_tattoo
			left_icon = root
			add_trait = volcano_tattoo
		}
	}
	else_if = {
		limit = { scope:tattoo = flag:cloud}
		send_interface_toast = {
			title = msg_added_tattoo
			left_icon = root
			add_trait = cloud_tattoo
		}
	}
	else_if = {
		limit = { scope:tattoo = flag:vine}
		send_interface_toast = {
			title = msg_added_tattoo
			left_icon = root
			add_trait = vine_tattoo
		}
	}
	else_if = {
		limit = { scope:tattoo = flag:void_dragon}
		send_interface_toast = {
			title = msg_added_tattoo
			left_icon = root
			add_trait = void_dragon_tattoo
		}
		random_list = {
			50 = {
				show_as_tooltip = { add_trait = ill }
				#TODO fix it
				custom_tooltip = pilgrimage.2009.a.tt
				hidden_effect = {
					#Adds the trait, sends event "health.2201" to those who care if health is brought too low
					contract_disease_effect = {
						DISEASE = ill
						TREATMENT_EVENT = no
					}
				}	
			}
			30 = {
				trigger = {
					has_trait = lunatic
				}
				death = {
					death_reason = death_accident
				}
			}
			25 = {
				trigger = {
					NOT = {
						has_trait = lunatic
					}
				}
				add_trait = lunatic
			}
			20 = {
				#Nothing happens
			}
			5 = {
				trigger = {
					NOT = {
						has_trait = lunatic
					}
				}
				death = { 
					death_reason = death_accident
				}
			}
		}
	}
	## Always 50% chance of being wounded on accident
	random = {
		chance = 50
		increase_wounds_effect = { REASON = accident }
	}
}
