# Gain Building
rokugan_building.0003 = {
	type = character_event
	title = rokugan_building.0003.t

	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { exists = scope:previous_owner }
				desc = rokugan_building.0003.with_owner.desc
			}
			desc = rokugan_building.0003.without_owner.desc
		}
	}

	theme = stewardship_wealth_focus	
	left_portrait = {
		character = root
		animation = personality_greedy
	}
	override_background = { reference = throne_room }

	option = {
		name = rokugan_building.0003.a
		rokugan_building_add_modifier_effect = yes
	}

	after = {
		hidden_effect = {
			scope:temp_character = {
				death = { death_reason = death_vanished }
			}
		}
	}
}

# Lose Building
rokugan_building.0004 = {
	type = character_event
	title = rokugan_building.0004.t
	desc = rokugan_building.0004.desc
	theme = stewardship_wealth_focus
	left_portrait = {
		character = root
		animation = personality_vengeful
	}
	override_background = { reference = burning_building }

	option = {
		name = rokugan_building.0004.a
		rokugan_building_remove_modifier_effect = yes
	}

	after = {
		hidden_effect = {
			scope:temp_character = {
				death = { death_reason = death_vanished }
			}
		}
	}
}

# Gain Building (Setup)
rokugan_building.0005 = {
	hidden = yes

	immediate = {
		random_province = {
			save_scope_as = birthplace
		}
		create_character = {
			gender = root
			template_character = root
			location = scope:birthplace
			save_scope_as = temp_character
		}
		scope:temp_character = {
			set_variable = {
				name  = building_name
				value = scope:building_name
			}
		}
		trigger_event = rokugan_building.0003
	}
}
